	$('#GotoTop').click(function(){
		$('html,body').animate({scrollTop:0});
	});
	$(window).scroll(function(){
		if($(window).scrollTop()>100){
			$('.returnBtn').css({'display':"block"})
		}
		else{
			$('.returnBtn').css({'display':"none"})
		}
	});
});
